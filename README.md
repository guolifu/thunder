6.视图机制优化，支持跨控制器，跨模块输出页面<br>

demo:<br>

namespace home\controller;<br>
use \thunder\View;<br>
class StudentController{<br>
    　use View;<br>
    　public function index(){<br>
        　　　$this->display();<br>
        　　　$this->display('index1');<br>
        　　　$this->display('index/url');<br>
        　　　$this->display('api/index/views');<br>
    　}<br>
}<br>
